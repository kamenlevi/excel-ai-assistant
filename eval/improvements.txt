For updating a comment, use:
```javascript
const sheet = workbook.worksheets.getActiveWorksheet();
const commentRange = sheet.getRange("A1");
commentRange.load("comment");
await context.sync();
if (commentRange.comment) {
  commentRange.comment.text = "Updated comment " + new Date().toLocaleDateString();
} else {
  commentRange.comment = { text: "New comment " + new Date().toLocaleDateString() };
}
await context.sync();
```
