To improve the handling of conditional formatting, add the following example:
```javascript
// Highlight cells in the Score column red if below 50
const sheet = workbook.worksheets.getActiveWorksheet();
const used = sheet.getUsedRange();
used.load("values");
await context.sync();
const rows = used.values;
const hdr = rows[0].map(h => String(h).toLowerCase().trim());
const col = hdr.indexOf("score");
if (col === -1) throw new Error("Column 'Score' not found.");
const conditionalFormat = used.conditionalFormats.add(Excel.ConditionalFormatType.cellValue);
conditionalFormat.formula = `=${used.columns[col].address} < 50`;
conditionalFormat.font.color = "#FF0000";
await context.sync();
```

For filtering, include this rule:
```javascript
// Filter the Department column to show only Engineering
const sheet = workbook.worksheets.getActiveWorksheet();
const used = sheet.getUsedRange();
used.load("values");
await context.sync();
const hdr = used.values[0].map(h => String(h).toLowerCase().trim());
const col = hdr.indexOf("department");
if (col === -1) throw new Error("Column 'Department' not found.");
used.autoFilter.apply({ column: col + 1, criteria1: "Engineering" });
await context.sync();
```

// Add a sparkline to cell D2 to display the high and low points of the data in cells B2:B10
